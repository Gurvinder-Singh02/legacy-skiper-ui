## Usage

```tsx
import { ExpandedTabs } from "../ui/expanded-tabs"
```
```tsx

const tabs = [
    { title: "Dashboard", icon: Home },
    { title: "Notifications", icon: Bell },
    { type: "separator" as const },
    { title: "Settings", icon: Settings },
    { title: "Support", icon: HelpCircle },
    { title: "Security", icon: Shield },
  ]
```

```tsx
function BasicExample() {
  return (
     <div className="flex items-center justify-center">
      <ExpandedTabs tabs={tabs} />
      <ExpandedTabs tabs={tabs} activeColor="text-blue-500" className="border-blue-200 dark:border-blue-800" />
    </div>
  )
}
```

## Usage 
define tabs object
```bash
import { Home,Bell,Settings,HelpCircle, Shield, User, } from "lucide-react"
const tabs = [
    { title: "Dashboard", icon: Home },
    { title: "Notifications", icon: Bell },
    { type: "separator" as const },
    { title: "Settings", icon: Settings },
    { title: "Support", icon: HelpCircle },
    { title: "Security", icon: Shield },
  ]
```

Use the component
```bash
 <ExpandedTabs tabs={tabs} />
```
